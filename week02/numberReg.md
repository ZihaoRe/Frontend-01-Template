```js
//decimal
//整数
/^0|[1-9]\d*$/
//小数
/^\.\d+|(0|[1-9]\d*)\.?\d*$/
//科学计数
/^(\.\d+|(0|[1-9]\d*)\.?\d*)([eE][\+\-]?\d+)?$/

//hex
/^0[xX][0-9a-fA-F]+$/

//octal
/^0[oO][0-7]+$/

//binary
/^0[bB][01]+$/
```
